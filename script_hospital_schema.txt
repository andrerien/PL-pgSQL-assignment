-- Script SQL para modelar um sistema de gestão de hospital no PostgreSQL

-- Tabela de cidades
CREATE TABLE cidade (
    cidade_id INT NOT NULL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    uf CHAR(2) NOT NULL,
    UNIQUE (nome, uf)
);

ALTER TABLE cidade
 ALTER COLUMN cidade_id ADD GENERATED BY DEFAULT AS IDENTITY;

-- Tabela de pacientes
CREATE TABLE paciente (
    paciente_id INT NOT NULL PRIMARY KEY,
    nome_completo VARCHAR(255) NOT NULL,
    data_nascimento DATE NOT NULL,
    sexo CHAR(3) NOT NULL,
    cpf CHAR(14) NOT NULL,
    telefone VARCHAR(20) NOT NULL,
    cep CHAR(10),
    endereco VARCHAR(255),
    complemento VARCHAR(100),
    cidade_id INT NOT NULL,
    FOREIGN KEY (cidade_id) REFERENCES cidade(cidade_id)
);

ALTER TABLE paciente
 ALTER COLUMN paciente_id ADD GENERATED BY DEFAULT AS IDENTITY;

CREATE TABLE especialidade (
    especialidade_id INT NOT NULL PRIMARY KEY,
    nome VARCHAR(255) NOT NULL
);

ALTER TABLE especialidade
 ALTER COLUMN especialidade_id ADD GENERATED BY DEFAULT AS IDENTITY;

-- Tabela de médicos
CREATE TABLE medico (
    medico_id INT NOT NULL PRIMARY KEY,
    crm_numero VARCHAR(10) NOT NULL DEFAULT '00000',
    crm_uf CHAR(2) NOT NULL DEFAULT 'BR',
    nome_completo VARCHAR(255) NOT NULL, 
    telefone VARCHAR(20) NOT NULL,
    email VARCHAR(255),
    especialidade_id INT NOT NULL,
    FOREIGN KEY (especialidade_id) REFERENCES especialidade(especialidade_id)
);

ALTER TABLE medico
 ALTER COLUMN medico_id ADD GENERATED BY DEFAULT AS IDENTITY;

CREATE TABLE medicamento (
  medicamento_id INT NOT NULL PRIMARY KEY,
  principio_ativo VARCHAR(255),
  cnpj VARCHAR(18),
  laboratorio VARCHAR(160) NOT NULL,
  codggrem VARCHAR(18),
  ean VARCHAR(18),
  nome VARCHAR(255) NOT NULL,
  apresentacao VARCHAR(255) NOT NULL,
  preco_fabricante NUMERIC(8,2) DEFAULT NULL,
  preco_comercial NUMERIC(8,2) DEFAULT NULL,
  restricao_hospitalar CHAR(3) DEFAULT NULL
);

ALTER TABLE medicamento
 ALTER COLUMN medicamento_id ADD GENERATED BY DEFAULT AS IDENTITY;

-- Tabela de consultas
CREATE TABLE consulta (
    consulta_id INT NOT NULL PRIMARY KEY,
    paciente_id INT NOT NULL,
    medico_id INT NOT NULL,
    data_agendamento TIMESTAMP NOT NULL,
    data_execucao TIMESTAMP,
    diagnostico TEXT,
    possui_prescricao INT NOT NULL DEFAULT 0,
    UNIQUE (consulta_id, paciente_id, medico_id),
    FOREIGN KEY (paciente_id) REFERENCES paciente(paciente_id),
    FOREIGN KEY (medico_id) REFERENCES medico(medico_id)
);

ALTER TABLE consulta
 ALTER COLUMN consulta_id ADD GENERATED BY DEFAULT AS IDENTITY;

CREATE TABLE prescricao (
    medicamento_id INT NOT NULL,
    consulta_id INT NOT NULL,
    data_prescricao TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    periodo INT NOT NULL, -- Número de dias para os quais o medicamento é prescrito
    posologia TEXT NOT NULL,
    quantidade TEXT NOT NULL,  
    PRIMARY KEY (medicamento_id, consulta_id),
    FOREIGN KEY (medicamento_id) REFERENCES medicamento(medicamento_id),
    FOREIGN KEY (consulta_id) REFERENCES consulta(consulta_id)
);
 
-- Tabela de leitos
CREATE TABLE leito (
    leito_id INT PRIMARY KEY,
    quarto INT NOT NULL,
    andar INT NOT NULL,
    bloco INT NOT NULL,
    status_ocupacao BOOLEAN DEFAULT FALSE
);

ALTER TABLE leito
 ALTER COLUMN leito_id ADD GENERATED BY DEFAULT AS IDENTITY;

-- Tabela de internações
CREATE TABLE internacao (
    internacao_id INT NOT NULL PRIMARY KEY,
    paciente_id INT NOT NULL,
    leito_id INT NOT NULL,
    data_internacao TIMESTAMP NOT NULL,
    data_liberacao TIMESTAMP,
    UNIQUE (paciente_id, leito_id, data_internacao),
    FOREIGN KEY (paciente_id) REFERENCES paciente(paciente_id),
    FOREIGN KEY (leito_id) REFERENCES leito(leito_id)
);

ALTER TABLE internacao
 ALTER COLUMN internacao_id ADD GENERATED BY DEFAULT AS IDENTITY;

CREATE TABLE medicamento_controle (
    medicamento_controle_id INT NOT NULL PRIMARY KEY,
    medicamento_id INT NOT NULL,
    data_modificacao TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    preco_comercial_old NUMERIC NOT NULL DEFAULT 0.0,
    preco_comercial_new NUMERIC NOT NULL DEFAULT 0.0,
    FOREIGN KEY (medicamento_id) REFERENCES medicamento(medicamento_id)
);

ALTER TABLE medicamento_controle
 ALTER COLUMN medicamento_controle_id ADD GENERATED BY DEFAULT AS IDENTITY;
